//TireMaster 5000// #include #include #include #include // ======================== // CONFIGURATION SETTINGS // ======================== #define DEBUG true // Set to false to disable serial output // Timing Intervals (milliseconds) const unsigned long PRESSURE_READ_INTERVAL = 1000; // 1 second const unsigned long STABILITY_CHECK_INTERVAL = 1000; // 1 second const unsigned long LCD_UPDATE_INTERVAL = 3000; // 3 seconds const unsigned long LCD_RESET_INTERVAL = 300000; // 5 minutes const unsigned long DEBOUNCE_INTERVAL = 50; // 50ms const unsigned long STARTUP_DELAY = 10000; // 10 seconds const unsigned long STABLE_TIMEOUT = 30000; // 30 seconds const unsigned long ALERT_DURATION = 1500; // 1.5 seconds const unsigned long PAUSE_DURATION = 5000; // 5 seconds // Cycle Configuration const unsigned long SHORT_CYCLE_DURATION = 30000; // 30 seconds const unsigned long MEDIUM_CYCLE_DURATION = 60000; // 1 minute const unsigned long LONG_CYCLE_DURATION = 120000; // 2 minutes const unsigned long VERY_LONG_CYCLE_DURATION = 180000;// 3 minutes const int VERY_LONG_THRESHOLD = 50; // 50+ PSI difference const int LONG_THRESHOLD = 40; // 40+ PSI difference const int MEDIUM_THRESHOLD = 10; // 10+ PSI difference // Pressure Settings const int TARGET_PSI_OPTIONS[] = {95, 70, 40, 20}; // Available target values const int PRESSURE_HYSTERESIS = 5; // +/- PSI tolerance const int PRESSURE_SENSOR_MIN = 102; // Analog at 0 PSI const int PRESSURE_SENSOR_MAX = 921; // Analog at 100 PSI const int PRESSURE_MAPPING_MIN = 0; // Mapped minimum PSI const int PRESSURE_MAPPING_MAX = 100; // Mapped maximum PSI // Hardware Configuration const int INFLATE_RELAY_PIN = 6; const int DEFLATE_RELAY_PIN = 7; const int ALERT_RELAY_PIN = 8; const int TOGGLE_UP_PIN = A1; const int TOGGLE_DOWN_PIN = A2; const int PRESSURE_SENSOR_PIN = A0; // LCD Configuration const int LCD_I2C_ADDRESS = 0x27; const int LCD_COLUMNS = 16; const int LCD_ROWS = 2; // ======================== // SYSTEM VARIABLES // ======================== LiquidCrystal_I2C lcd(LCD_I2C_ADDRESS, LCD_COLUMNS, LCD_ROWS); Bounce debouncerUp; Bounce debouncerDown; // Timing variables unsigned long lastPressureReadTime = 0; unsigned long lastStabilityCheckTime = 0; unsigned long operationStartTime = 0; unsigned long pauseStartTime = 0; // Added missing declaration unsigned long targetTimerStart = 0; unsigned long stableStartTime = 0; unsigned long alertStartTime = 0; unsigned long lastLCDReset = 0; unsigned long lastLCDUpdate = 0; unsigned long operationDuration = 0; unsigned long targetTimerDuration = 0; // Pressure variables int currentPressure = 0; int targetPressure = TARGET_PSI_OPTIONS[0]; int psiIndex = 0; // State flags bool inflating = false; bool deflating = false; bool isPausing = false; bool monitoring = false; bool isStable = false; // LCD buffers char lcdBufferLine1[LCD_COLUMNS + 1]; char lcdBufferLine2[LCD_COLUMNS + 1]; // ======================== // SETUP FUNCTION // ======================== void setup() { wdt_enable(WDTO_8S); #if DEBUG Serial.begin(9600); Serial.println(F("\n=== TireMaster 5000 Initializing ===")); Serial.print(F("Initial Target Pressure: ")); Serial.println(targetPressure); Serial.println(F("Cycle Durations:")); Serial.print(F(" Short: ")); Serial.print(SHORT_CYCLE_DURATION/1000); Serial.println(F("s")); Serial.print(F(" Medium: ")); Serial.print(MEDIUM_CYCLE_DURATION/1000); Serial.println(F("s")); Serial.print(F(" Long: ")); Serial.print(LONG_CYCLE_DURATION/1000); Serial.println(F("s")); Serial.print(F(" Very Long: ")); Serial.print(VERY_LONG_CYCLE_DURATION/1000); Serial.println(F("s")); #endif // Initialize hardware pinMode(INFLATE_RELAY_PIN, OUTPUT); pinMode(DEFLATE_RELAY_PIN, OUTPUT); pinMode(ALERT_RELAY_PIN, OUTPUT); digitalWrite(INFLATE_RELAY_PIN, HIGH); digitalWrite(DEFLATE_RELAY_PIN, HIGH); digitalWrite(ALERT_RELAY_PIN, HIGH); // Initialize LCD lcd.init(); lcd.backlight(); lcd.clear(); lcd.setCursor(0, 0); lcd.print(F("TireMaster 5000")); lcd.setCursor(0, 1); lcd.print(F("Initializing...")); // Initialize buttons pinMode(TOGGLE_UP_PIN, INPUT_PULLUP); pinMode(TOGGLE_DOWN_PIN, INPUT_PULLUP); debouncerUp.attach(TOGGLE_UP_PIN); debouncerUp.interval(DEBOUNCE_INTERVAL); debouncerDown.attach(TOGGLE_DOWN_PIN); debouncerDown.interval(DEBOUNCE_INTERVAL); // Initialize LCD buffers memset(lcdBufferLine1, ' ', LCD_COLUMNS); memset(lcdBufferLine2, ' ', LCD_COLUMNS); lcdBufferLine1[LCD_COLUMNS] = '\0'; lcdBufferLine2[LCD_COLUMNS] = '\0'; // Startup sequence unsigned long startMillis = millis(); while (millis() - startMillis < STARTUP_DELAY) { wdt_reset(); debouncerUp.update(); debouncerDown.update(); if (debouncerUp.fell()) { psiIndex = (psiIndex + 1) % 4; targetPressure = TARGET_PSI_OPTIONS[psiIndex]; #if DEBUG Serial.print(F("New target pressure: ")); Serial.println(targetPressure); #endif } if (debouncerDown.fell()) { psiIndex = (psiIndex - 1 + 4) % 4; targetPressure = TARGET_PSI_OPTIONS[psiIndex]; #if DEBUG Serial.print(F("New target pressure: ")); Serial.println(targetPressure); #endif } char tempBuffer[LCD_COLUMNS + 1]; snprintf(tempBuffer, sizeof(tempBuffer), "Target: %d %lus ", targetPressure, (STARTUP_DELAY - (millis() - startMillis)) / 1000); strncpy(lcdBufferLine1, tempBuffer, LCD_COLUMNS); int sensorValue = analogRead(PRESSURE_SENSOR_PIN); currentPressure = map(sensorValue, PRESSURE_SENSOR_MIN, PRESSURE_SENSOR_MAX, PRESSURE_MAPPING_MIN, PRESSURE_MAPPING_MAX); snprintf(tempBuffer, sizeof(tempBuffer), "Current: %d PSI ", currentPressure); strncpy(lcdBufferLine2, tempBuffer, LCD_COLUMNS); lcd.setCursor(0, 0); lcd.print(lcdBufferLine1); lcd.setCursor(0, 1); lcd.print(lcdBufferLine2); delay(100); } targetTimerStart = millis(); setTargetTimerDuration(); stableStartTime = millis(); #if DEBUG Serial.println(F("System initialization complete!")); Serial.println(F("Entering main loop...")); #endif } // ======================== // MAIN LOOP // ======================== void loop() { wdt_reset(); unsigned long currentMillis = millis(); // Update inputs debouncerUp.update(); debouncerDown.update(); // Handle pressure changes with 5-second pause if (debouncerUp.fell() || debouncerDown.fell()) { if (debouncerUp.fell()) { psiIndex = (psiIndex + 1) % 4; targetPressure = TARGET_PSI_OPTIONS[psiIndex]; #if DEBUG Serial.print(F("Target pressure increased to: ")); Serial.println(targetPressure); #endif } else { psiIndex = (psiIndex - 1 + 4) % 4; targetPressure = TARGET_PSI_OPTIONS[psiIndex]; #if DEBUG Serial.print(F("Target pressure decreased to: ")); Serial.println(targetPressure); #endif } stopAll(); isPausing = true; pauseStartTime = currentMillis; monitoring = false; targetTimerStart = currentMillis; isStable = false; setTargetTimerDuration(); lastLCDUpdate = 0; #if DEBUG Serial.println(F("Starting 5-second pause after pressure change")); #endif } // LCD maintenance if (currentMillis - lastLCDReset >= LCD_RESET_INTERVAL) { lcd.init(); delay(50); lcd.backlight(); lcd.clear(); memset(lcdBufferLine1, ' ', LCD_COLUMNS); memset(lcdBufferLine2, ' ', LCD_COLUMNS); lastLCDUpdate = 0; lastLCDReset = currentMillis; } // Read pressure sensor if (currentMillis - lastPressureReadTime >= PRESSURE_READ_INTERVAL) { int sensorValue = analogRead(PRESSURE_SENSOR_PIN); currentPressure = map(sensorValue, PRESSURE_SENSOR_MIN, PRESSURE_SENSOR_MAX, PRESSURE_MAPPING_MIN, PRESSURE_MAPPING_MAX); lastPressureReadTime = currentMillis; #if DEBUG Serial.print(F("Pressure: ")); Serial.print(currentPressure); Serial.print(F(" PSI | Target: ")); Serial.print(targetPressure); Serial.print(F("")); Serial.print(PRESSURE_HYSTERESIS); Serial.println(F(" PSI")); #endif } // Check system stability if (currentMillis - lastStabilityCheckTime >= STABILITY_CHECK_INTERVAL) { if (!inflating && !deflating && !isPausing) { if (currentPressure >= targetPressure - PRESSURE_HYSTERESIS && currentPressure <= targetPressure + PRESSURE_HYSTERESIS) { if (!isStable) { stableStartTime = currentMillis; isStable = true; #if DEBUG Serial.println(F("System entered stable state")); #endif } } else { if (isStable && (currentMillis - stableStartTime >= STABLE_TIMEOUT)) { triggerAlarm(); isStable = false; } isStable = false; } } lastStabilityCheckTime = currentMillis; } // Main control logic if (!monitoring && !isPausing) { if (!inflating && !deflating) { int pressureDifference = abs(currentPressure - targetPressure); if (currentPressure < targetPressure - PRESSURE_HYSTERESIS) { // Determine inflation duration if (pressureDifference >= VERY_LONG_THRESHOLD) { operationDuration = VERY_LONG_CYCLE_DURATION; #if DEBUG Serial.print(F("Starting VERY LONG inflation (")); #endif } else if (pressureDifference >= LONG_THRESHOLD) { operationDuration = LONG_CYCLE_DURATION; #if DEBUG Serial.print(F("Starting LONG inflation (")); #endif } else if (pressureDifference >= MEDIUM_THRESHOLD) { operationDuration = MEDIUM_CYCLE_DURATION; #if DEBUG Serial.print(F("Starting MEDIUM inflation (")); #endif } else { operationDuration = SHORT_CYCLE_DURATION; #if DEBUG Serial.print(F("Starting SHORT inflation (")); #endif } #if DEBUG Serial.print(operationDuration/1000); Serial.println(F("s)")); #endif startInflation(); operationStartTime = currentMillis; } else if (currentPressure > targetPressure + PRESSURE_HYSTERESIS) { // Determine deflation duration if (pressureDifference >= VERY_LONG_THRESHOLD) { operationDuration = VERY_LONG_CYCLE_DURATION; #if DEBUG Serial.print(F("Starting VERY LONG deflation (")); #endif } else if (pressureDifference >= LONG_THRESHOLD) { operationDuration = LONG_CYCLE_DURATION; #if DEBUG Serial.print(F("Starting LONG deflation (")); #endif } else if (pressureDifference >= MEDIUM_THRESHOLD) { operationDuration = MEDIUM_CYCLE_DURATION; #if DEBUG Serial.print(F("Starting MEDIUM deflation (")); #endif } else { operationDuration = SHORT_CYCLE_DURATION; #if DEBUG Serial.print(F("Starting SHORT deflation (")); #endif } #if DEBUG Serial.print(operationDuration/1000); Serial.println(F("s)")); #endif startDeflation(); operationStartTime = currentMillis; } } // Time-based termination if (inflating && (currentMillis - operationStartTime >= operationDuration)) { stopAll(); isPausing = true; pauseStartTime = currentMillis; #if DEBUG Serial.print(F("Inflation completed after ")); Serial.print((currentMillis - operationStartTime)/1000); Serial.println(F("s")); #endif } if (deflating && (currentMillis - operationStartTime >= operationDuration)) { stopAll(); isPausing = true; pauseStartTime = currentMillis; #if DEBUG Serial.print(F("Deflation completed after ")); Serial.print((currentMillis - operationStartTime)/1000); Serial.println(F("s")); #endif } } // End pause period if (isPausing && (currentMillis - pauseStartTime >= PAUSE_DURATION)) { isPausing = false; #if DEBUG Serial.println(F("Pause period ended")); #endif } // Handle target timer expiration if (targetTimerDuration > 0 && currentMillis - targetTimerStart >= targetTimerDuration) { targetPressure = 95; psiIndex = 0; targetTimerDuration = 0; isStable = false; monitoring = false; digitalWrite(ALERT_RELAY_PIN, HIGH); startInflation(); #if DEBUG Serial.println(F("Target timer expired - resetting to 95 PSI")); #endif } // Update LCD if (currentMillis - lastLCDUpdate >= LCD_UPDATE_INTERVAL) { char status[10]; if (inflating) strcpy(status, "Inflating"); else if (deflating) strcpy(status, "Deflating"); else if (isPausing) strcpy(status, "Pausing"); else if (isStable) strcpy(status, "Stable"); else strcpy(status, "Monitor"); char newLine1[LCD_COLUMNS + 1]; snprintf(newLine1, sizeof(newLine1), "T: %d %s", targetPressure, status); padToWidth(newLine1, LCD_COLUMNS); char newLine2[LCD_COLUMNS + 1]; if (targetTimerDuration > 0) { unsigned long remaining = targetTimerDuration - (currentMillis - targetTimerStart); if (remaining > targetTimerDuration) remaining = 0; char timeStr[6]; formatTime(timeStr, remaining); snprintf(newLine2, sizeof(newLine2), "C: %d PSI %s", currentPressure, timeStr); } else { snprintf(newLine2, sizeof(newLine2), "C: %d PSI", currentPressure); } padToWidth(newLine2, LCD_COLUMNS); if (strcmp(lcdBufferLine1, newLine1) != 0 || strcmp(lcdBufferLine2, newLine2) != 0) { strncpy(lcdBufferLine1, newLine1, LCD_COLUMNS); strncpy(lcdBufferLine2, newLine2, LCD_COLUMNS); lcd.setCursor(0, 0); lcd.print(lcdBufferLine1); lcd.setCursor(0, 1); lcd.print(lcdBufferLine2); } lastLCDUpdate = currentMillis; } } // ======================== // HELPER FUNCTIONS // ======================== void setTargetTimerDuration() { switch (targetPressure) { case 70: targetTimerDuration = 3600000UL; break; // 1 hour case 40: targetTimerDuration = 1800000UL; break; // 30 minutes case 20: targetTimerDuration = 1800000UL; break; // 30 minutes default: targetTimerDuration = 0; } #if DEBUG Serial.print(F("Target timer set to: ")); char timeStr[6]; formatTime(timeStr, targetTimerDuration); Serial.println(timeStr); #endif } void startInflation() { inflating = true; deflating = false; digitalWrite(INFLATE_RELAY_PIN, LOW); digitalWrite(DEFLATE_RELAY_PIN, HIGH); #if DEBUG Serial.println(F("INFLATION STARTED")); #endif } void startDeflation() { inflating = false; deflating = true; digitalWrite(INFLATE_RELAY_PIN, HIGH); digitalWrite(DEFLATE_RELAY_PIN, LOW); #if DEBUG Serial.println(F("DEFLATION STARTED")); #endif } void stopAll() { digitalWrite(INFLATE_RELAY_PIN, HIGH); digitalWrite(DEFLATE_RELAY_PIN, HIGH); inflating = false; deflating = false; #if DEBUG Serial.println(F("All operations stopped")); #endif } void triggerAlarm() { lcd.setCursor(0, 0); lcd.print(F("BROKEN HOSE ")); lcd.setCursor(0, 1); lcd.print(F("CHECK SYSTEM ")); #if DEBUG Serial.println(F("ALARM TRIGGERED - Check system!")); #endif for (int i = 0; i < 2; i++) { digitalWrite(ALERT_RELAY_PIN, LOW); delay(ALERT_DURATION); digitalWrite(ALERT_RELAY_PIN, HIGH); delay(500); } memset(lcdBufferLine1, ' ', LCD_COLUMNS); memset(lcdBufferLine2, ' ', LCD_COLUMNS); } void triggerAlert() { digitalWrite(ALERT_RELAY_PIN, LOW); delay(500); digitalWrite(ALERT_RELAY_PIN, HIGH); #if DEBUG Serial.println(F("ALERT - Pressure out of range")); #endif } void padToWidth(char* str, int width) { int len = strlen(str); while (len < width) { str[len] = ' '; len++; } str[width] = '\0'; } void formatTime(char* buffer, unsigned long ms) { unsigned long minutes = ms / 60000; unsigned long seconds = (ms % 60000) / 1000; snprintf(buffer, 6, "%lu:%02lu", minutes, seconds); }
